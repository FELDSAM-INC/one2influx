def initialize(xml, client)

def serialize_as_points
  points = []
  @metrics.each do |metric_name, metric_value|
    points << {
        :name => metric_name,
        :tags => @tags,
        :fields => {
            :value => metric_value.to_f
        }
    }
  end

  return points
end
